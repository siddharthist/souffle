46 /**
47  * @class MaxMatching
48  * @Brief Computes a maximum matching with Hopcroft-Karp algorithm
49  *
50  * This class is a helper class for RamIndexAnalysis.
51  *
52  * This implements a standard maximum matching algorithm for a bi-partite graph
53  * also known as a marriage problem. Given a set of edges in a bi-partite graph
54  * select a subset of edges that each node in the bi-partite graph has at most
55  * one adjacent edge associated with.
56  *
57  * The nodes of the bi-partite graph represent index-signatures stemming from
58  * RAM operations and RAM existence checks for a relation. A relation between
59  * two nodes represent whether an index operation subsumes another operation.
60  *
61  * Source: http://en.wikipedia.org/wiki/Hopcroft%E2%80%93Karp_algorithm#Pseudocode
62  */
63 class MaxMatching {
64 public:
65  /* The nodes of the bi-partite graph are index signatures of RAM operation */
66  using Nodes = std::set<SearchSignature, std::greater<SearchSignature>>;
67 
68  /**
69  * Matching represent a solution of the matching, i.e., which node in the bi-partite
70  * graph maps to another node. If no map exist for a node, there is no adjacent edge
71  * exists for that node.
72  */
73  using Matchings = std::map<SearchSignature, SearchSignature, std::greater<SearchSignature>>;
74 
75 public:
76  /**
77  * @Brief solve the maximum matching problem
78  * @result returns the matching
79  */
80  const Matchings& solve();
81 
82  /**
83  * @Brief get number of matches in the solution
84  * @return number of matches
85  */
86  int getNumMatchings() const {
87  return match.size() / 2;
88  }

91  * @Brief add an edge to the bi-partite graph
92  * @param u search signature

95  void addEdge(SearchSignature u, SearchSignature v);

97 protected:
98  /**
99  * @Brief get match for a search signature
100  * @param v search signature
101  */

117  */

122  * Edges in the bi-partite graph
123  */

220  /** @Brief insert a total order index
221  * @param size of the index
222  */
223  void insertDefaultTotalIndex(size_t arity) {
224  Chain chain = std::set<SearchSignature>();
225  SearchSignature fullIndexKey = (1 << arity) - 1;

241  /** @Brief count the number of bits in key */
242  static size_t card(SearchSignature cols) {
243  size_t sz = 0, idx = 1;
244  for (size_t i = 0; i < sizeof(SearchSignature) * 8; i++) {
245  if (idx & cols) {
246  sz++;
247  }
248  idx *= 2;
249  }
250  return sz;
251  }

253  /** @Brief maps search columns to an lexicographical order (labeled by a number) */
254  int map(SearchSignature cols) const {
255  assert(orders.size() == chainToOrder.size() && "Order and Chain Sizes do not match!!");
256  int i = 0;
257  for (auto it = chainToOrder.begin(); it != chainToOrder.end(); ++it, ++i) {
258  if (it->find(cols) != it->end()) {
259  assert((size_t)i < orders.size());
260  return i;
261  }
262  }
263  std::cerr << "Cannot find matching lexicographical order" << std::endl;
264  abort();
265  }

273  /** @Brief insert an index based on the delta */
274  void insertIndex(LexOrder& ids, SearchSignature delta) {
275  int pos = 0;
276  SearchSignature mask = 0;
277 
278  while (mask < delta) {
279  mask = SearchSignature(1 << (pos));
280  SearchSignature result = (delta) & (mask);
281  if (result) {
282  ids.push_back(pos);
283  }
284  pos++;
285  }
286  }

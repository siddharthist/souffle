446 /**
447  * @brief Convert terms of a conjunction to a list
448  * @param A RAM condition
449  * @param A list of RAM conditions
450  *
451  * Convert a condition of the format C1 /\ C2 /\ ... /\ Cn
452  * to a list {C1, C2, ..., Cn}.
453  */
454 inline std::vector<std::unique_ptr<RamCondition>> toConjunctionList(const RamCondition* condition) {
455  std::vector<std::unique_ptr<RamCondition>> list;
456  std::stack<const RamCondition*> stack;
457  if (condition != nullptr) {
458  stack.push(condition);
459  while (!stack.empty()) {
460  condition = stack.top();
461  stack.pop();
462  if (const auto* ramConj = dynamic_cast<const RamConjunction*>(condition)) {
463  stack.push(&ramConj->getLHS());
464  stack.push(&ramConj->getRHS());
465  } else {
466  list.emplace_back(condition->clone());
467  }
468  }
469  }
470  return list;
471 }

473 /**
474  * @brief Convert list of conditions to a conjunction
475  * @param A list of RAM conditions
476  * @param A RAM condition
477  *
478  * Convert a list {C1, C2, ..., Cn} to a condition of
479  * the format C1 /\ C2 /\ ... /\ Cn.
480  */
481 inline std::unique_ptr<RamCondition> toCondition(const std::vector<const RamCondition*>& list) {
482  std::unique_ptr<RamCondition> result;
483  for (const RamCondition* cur : list) {
484  if (result == nullptr) {
485  result = std::unique_ptr<RamCondition>(cur->clone());
486  } else {
487  result = std::make_unique<RamConjunction>(
488  std::move(result), std::unique_ptr<RamCondition>(cur->clone()));
489  }
490  }
491  return result;
492 }

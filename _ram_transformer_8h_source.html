31 /**
32  * @Class RamTransformer
33  * @Brief abstract transformer class for a translation unit
34  *
35  * This is an abstract class to implement transformers. A
36  * transformer takes a translation unit and changes its
37  * state.
38  *
39  * Transformers can be composed using other transformers.
40  *
41  * For debugging purposes, a transformer has a name
42  * (this will show up in the debug report) and a
43  * protected method transform(), that performs the
44  * actual transformation.
45  *
46  * The method apply is used to call transform() and
47  * does the reporting of the debug information.
48  *
49  */
50

53  RamTransformer() = default;

79 /**
80  * @Class RamTransformerSequence
81  * @Brief Composite sequence transformer
82  *
83  * A sequence of transformations is applied to a translation unit
84  * sequentially. The last transformation decides the outcome whether
85  * the code has been changed.
86  *
87  */
88 class RamTransformerSequence : public RamTransformer {
89 public:
90  template <typename... Tfs>
91  RamTransformerSequence(std::unique_ptr<Tfs>&&... tf) : RamTransformerSequence() {
92  std::unique_ptr<RamTransformer> tmp[] = {std::move(tf)...};
93  for (auto& cur : tmp) {
94  transformers.emplace_back(std::move(cur));
95  }
96  for (const auto& cur : transformers) {
97  (void)cur;
98  assert(cur);
99  }
100  }
101  RamTransformerSequence() = default;
102  std::string getName() const override {
103  return "RamTransformerSequence";
104  }
105  bool transform(RamTranslationUnit& tU) override {
106  bool changed = false;
107  // The last transformer decides the status
108  // of the change flag.
109  // Note that for other semantics, new transformer
110  // sequence class needs to be introduced.
111  for (auto const& cur : transformers) {
112  changed = cur->apply(tU);
113  }
114  return changed;
115  }
116 
117 protected:
118  /** sequence of transformers */
119  std::vector<std::unique_ptr<RamTransformer>> transformers;
120 };

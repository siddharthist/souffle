115 /**
116  * @class HoistConditionsTransformer
117  * @brief Hosts conditions in a loop-nest to the most-outer/semantically-correct loop
118  *
119  * Hoists the conditions to the earliest point in the loop nest where their
120  * evaluation is still semantically correct.
121  *
122  * The transformations assumes that filter operations are stored verbose,
123  * i.e. a conjunction is expressed by two consecutive filter operations.
124  * For example ..
125  *
126  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
127  * QUERY
128  * ...
129  * IF C1 /\ C2 then
130  * ...
131  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
132  *
133  * should be rewritten / or produced by the translator as
134  *
135  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
136  * QUERY
137  * ...
138  * IF C1
139  * IF C2
140  * ...
141  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
142  *
143  * otherwise the levelling becomes imprecise, i.e., for both conditions
144  * the most outer-level is sought rather than considered separately.
145  *
146  * If there are transformers prior to hoistConditions() that introduce
147  * conjunction, another transformer is required that splits the
148  * filter operations. However, at the moment this is not necessary
149  * because the translator delivers already the right RAM format.
150  *
151  * TODO: break-up conditions while transforming so that this requirement
152  * is removed.
153  */
154 class HoistConditionsTransformer : public RamTransformer {
155 public:
156  std::string getName() const override {
157  return "HoistConditionsTransformer";
158  }
159 
160  /**
161  * @brief Hoist filter operations.
162  * @param program that is transformed
163  * @return Flag showing whether the program has been changed by the transformation
164  *
165  * There are two types of conditions in
166  * filter operations. The first type depends on tuples of
167  * RamTupleOperation operations. The second type are independent of
168  * tuple access. Both types of conditions will be hoisted to
169  * the most out-scope such that the program is still valid.
170  */
171  bool hoistConditions(RamProgram& program);
172 
173 protected:
174  RamLevelAnalysis* rla{nullptr};
175 
176  bool transform(RamTranslationUnit& translationUnit) override {
177  rla = translationUnit.getAnalysis<RamLevelAnalysis>();
178  return hoistConditions(*translationUnit.getProgram());
179  }
180 };

338  * (Choice)/(IndexChoice) operations
339 
340  * If there exists Scan/IndexScan operations in the RAM, and the
341  * variables are used in a subsequent Filter operation but no
342  * subsequent operation in the tree (up until and including

46  * RAM is transferred into an equivalent Bytecode Representation (LVMCode) by LVMGenerator.

75  /** Execute the subroutine */
76  void executeSubroutine(const std::string& name, const std::vector<RamDomain>& arguments,
77  std::vector<RamDomain>& returnValues, std::vector<bool>& returnErrors) override {
78  LVMContext ctxt;
79  ctxt.setReturnValues(returnValues);
80  ctxt.setReturnErrors(returnErrors);
81  ctxt.setArguments(arguments);
82 
83  if (subroutines.find(name) != subroutines.cend()) {
84  execute(subroutines.at(name), ctxt);
85  } else {
86  // Parse and cache the program
87  LVMGenerator generator(translationUnit.getSymbolTable(),
88  translationUnit.getProgram()->getSubroutine(name), relationEncoder);
89  subroutines.emplace(std::make_pair(name, generator.getCodeStream()));
90  execute(subroutines.at(name), ctxt);
91  }
92  }

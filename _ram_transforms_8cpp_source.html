36 
37  visitDepthFirst(program, [&](const RamQuery& query) {
38  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
39  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
40  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
41  const RamCondition* condition = &filter->getCondition();

43  if (conditionList.size() > 1) {

53  } else {
54  filters.emplace_back(std::make_unique<RamFilter>(
55  std::unique_ptr<RamCondition>(std::move(tempCond)),
56  std::move(filters.back())));
57  }
58  }
59  node = std::move(filters.back());
60  }
61  }
62  node->apply(makeLambdaRamMapper(filterRewriter));

64  };

108 bool HoistConditionsTransformer::hoistConditions(RamProgram& program) {
109  // flag to determine whether the RAM program has changed
110  bool changed = false;
111 
112  // helper for collecting conditions from filter operations
113  auto addCondition = [](std::unique_ptr<RamCondition> condition,
114  RamCondition* c) -> std::unique_ptr<RamCondition> {
115  if (condition == nullptr) {
116  return std::unique_ptr<RamCondition>(c);
117  } else {
118  return std::make_unique<RamConjunction>(std::move(condition), std::unique_ptr<RamCondition>(c));
119  }
120  };
121 
122  // insert a new filter
123  auto insertFilter = [](RamOperation* op, std::unique_ptr<RamCondition>& condition) {
124  op->apply(makeLambdaRamMapper([&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
125  if (nullptr != dynamic_cast<RamOperation*>(node.get())) {
126  return std::make_unique<RamFilter>(std::move(condition),
127  std::unique_ptr<RamOperation>(dynamic_cast<RamOperation*>(node.release())));
128  }
129  return node;
130  }));
131  };
132 
133  // hoist conditions to the most outer scope if they
134  // don't depend on RamTupleOperations
135  visitDepthFirst(program, [&](const RamQuery& query) {
136  std::unique_ptr<RamCondition> newCondition;
137  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
138  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
139  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
140  const RamCondition& condition = filter->getCondition();
141  // if filter condition is independent of any RamTupleOperation,
142  // delete the filter operation and collect condition
143  if (rla->getLevel(&condition) == -1) {
144  changed = true;
145  newCondition = addCondition(std::move(newCondition), condition.clone());
146  node->apply(makeLambdaRamMapper(filterRewriter));
147  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
148  }
149  }
150  node->apply(makeLambdaRamMapper(filterRewriter));
151  return node;
152  };
153  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
154  if (newCondition != nullptr) {
155  // insert new filter operation at outer-most level of the query
156  changed = true;
157  insertFilter((RamOperation*)&query, newCondition);
158  }
159  });
160 
161  // hoist conditions for each RamTupleOperation operation
162  visitDepthFirst(program, [&](const RamTupleOperation& search) {
163  std::unique_ptr<RamCondition> newCondition;
164  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
165  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
166  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
167  const RamCondition& condition = filter->getCondition();
168  // if filter condition matches level of RamTupleOperation,
169  // delete the filter operation and collect condition
170  if (rla->getLevel(&condition) == search.getTupleId()) {
171  changed = true;
172  newCondition = addCondition(std::move(newCondition), condition.clone());
173  node->apply(makeLambdaRamMapper(filterRewriter));
174  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
175  }
176  }
177  node->apply(makeLambdaRamMapper(filterRewriter));
178  return node;
179  };
180  const_cast<RamTupleOperation*>(&search)->apply(makeLambdaRamMapper(filterRewriter));
181  if (newCondition != nullptr) {
182  // insert new filter operation after the search operation
183  changed = true;
184  insertFilter((RamOperation*)&search, newCondition);
185  }
186  });
187  return changed;

213 std::unique_ptr<RamCondition> MakeIndexTransformer::constructPattern(
214  std::vector<std::unique_ptr<RamExpression>>& queryPattern, bool& indexable,
215  std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
216  // Remaining conditions which cannot be handled by an index
217  std::unique_ptr<RamCondition> condition;
218  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
219  if (condition != nullptr) {
220  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
221  } else {
222  condition = std::move(c);
223  }
224  };
225 
226  // Build query pattern and remaining condition
227  for (auto& cond : conditionList) {
228  size_t element = 0;
229  if (std::unique_ptr<RamExpression> value = getExpression(cond.get(), element, identifier)) {
230  if (queryPattern[element] == nullptr) {
231  indexable = true;
232  queryPattern[element] = std::move(value);
233  } else {
234  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ, std::move(value),
235  std::unique_ptr<RamExpression>(queryPattern[element]->clone())));
236  }
237  } else {
238  addCondition(std::move(cond));
239  }
240  }
241 
242  // Avoid null-pointers for condition and query pattern
243  if (condition == nullptr) {
244  condition = std::make_unique<RamTrue>();
245  }
246 
247  for (auto& p : queryPattern) {
248  if (p == nullptr) {
249  p = std::make_unique<RamUndefValue>();
250  }
251  }
252  return condition;
253 }

305  auto addCondition = [&](std::unique_ptr<RamCondition> c) {

307  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
308  } else {
309  condition = std::move(c);
310  }
311  };
312  for (std::size_t i = 0; i < rel.getArity(); i++) {
313  if (prevPattern[i] != nullptr) {

560 bool HoistAggregateTransformer::hoistAggregate(RamProgram& program) {
561  // flag to determine whether the RAM program has changed
562  bool changed = false;
563 
564  // hoist aggregate if doesn't depend on anything
565  visitDepthFirst(program, [&](const RamQuery& query) {
566  int currLevel = -1;
567  int oldLevel = -1;
568  bool hoist = false;
569  bool newIndex = false;
570 
571  // Fields for new aggregate
572  AggregateFunction newFun;
573  const RamRelation* newRef;
574  RamExpression* newExp;
575  RamCondition* newCond;
576  std::vector<RamExpression*> newPattern;
577 
578  // Tracking aggregates seen to determine whether to
579  // hoist or not
580  std::vector<int> aggIds;
581 
582  // Removing an aggregate that can be hoisted if it exists
583  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRemover =
584  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
585  // Checking that there exists an aggregate to hoist
586  // If so, remove the aggregate from the loop nest
587  if (const RamAggregate* agg = dynamic_cast<RamAggregate*>(node.get())) {
588  aggIds.push_back(agg->getTupleId());
589  // assuming that RamTupleOperations have tupleIds 0, 1, 2, ...
590  if (!hoist && rla->getLevel(agg) < agg->getTupleId() - 1) {
591  // If all tuple ops between the rla->getLevel(agg) and agg
592  // are aggregates, then we do not transform

651

653  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggAdder =
654  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {

659  // Finding the operation right before the RamTupleOperation

671  } else {

673  }

692  std::unique_ptr<RamCondition>(newCond), oldLevel);
693  }
694  }
695  }
696  node->apply(makeLambdaRamMapper(aggAdder));
697  return node;
698  };

701  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
702  if (nullptr != dynamic_cast<RamOperation*>(node.get())) {
703  if (newIndex) {

716  newFun, std::make_unique<RamRelationReference>(newRef),
717  std::unique_ptr<RamExpression>(newExp), std::unique_ptr<RamCondition>(newCond),
718  std::move(queryPattern), oldLevel);
719  } else {

723  newFun, std::make_unique<RamRelationReference>(newRef),

725  oldLevel);
726  }
727  }

729  };
730 
731  if (hoist) {

733  // insert at the top of the loop nest
734  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggTopAdder));
735  } else {

737  }
738  }
739  });
